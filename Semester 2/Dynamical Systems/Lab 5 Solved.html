from sympy import symbols, Eq, solve, Matrix
import numpy as np

x, y = symbols('x y')

# Define the system of equations
eq1 = Eq(2*x - x**2 - x*y, 0)
eq2 = Eq(-y + x*y, 0)

# Solve the system for equilibria
equilibria = solve([eq1, eq2], (x, y))
print("Equilibria:", equilibria)

# Define the functions for the system
f1 = 2*x - x**2 - x*y
f2 = -y + x*y

# Compute the Jacobian matrix
J = Matrix([[f1.diff(x), f1.diff(y)], [f2.diff(x), f2.diff(y)]])
print("Jacobian matrix:", J)

A1 = J.subs([(x, 0), (y, 0)])
A2 = J.subs([(x, 2), (y, 0)])
A3 = J.subs([(x, 1), (y, 1)])
print("Jacobian at (0, 0):", A1)
print("Jacobian at (2, 0):", A2)
print("Jacobian at (1, 1):", A3)

eigenvalues_A1 = A1.eigenvals()
eigenvalues_A2 = A2.eigenvals()
eigenvalues_A3 = A3.eigenvals()

print("Eigenvalues of A1:", eigenvalues_A1)
print("Eigenvalues of A2:", eigenvalues_A2)
print("Eigenvalues of A3:", eigenvalues_A3)

    solution_orbits.append(sol)

# Plot all solution orbits in the same graphic
combined_solution = sum([line(zip(sol[:, 0], sol[:, 1])) for sol in solution_orbits])
