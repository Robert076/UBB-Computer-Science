# EXERCISE 3. Sensitivity with respect to small errors is a feature of chaos
reset()
from sage.plot.line import Line
N=40
r = 4
x = zero_vector(RR, N+1)
x[0] = 0.67
y = zero_vector(RR, N+1)
y[0] = 0.670001

for i in range(1,N):
    x[i] = r*x[i-1]*(1-x[i-1])
    y[i] = r*y[i-1]-1*r*y[i-1]^2
        
lx = line(Line(range(N),x,{}),color='blue', marker='o')
ly = line(Line(range(N),y,{}),color='red', marker='o')
l = lx+ly

l.axes_labels(('$n$','$x_n$'))
show(l)

# EXERCISE 4. y'=2xy, y(0)=1, x\in [0,1]
reset()
from sage.plot.line import Line
import numpy

N = 11
h = 1/(N-1)
x = zero_vector(RR, N)
y = zero_vector(RR, N) # the Euler method
y[0] = 1
yy = zero_vector(RR, N) # improved Euler method
yy[0] = 1
ye = zero_vector(RR, N) # the exact solution is y(x) = e^(x^2)
ye[0] = 1

for i in range(1,N):
    x[i] = x[i-1]+h
    y[i] = y[i-1]+2*h*x[i-1]*y[i-1] # the Euler method
    yy[i] = yy[i-1] + 0.5*2*h*x[i-1]*y[i-1] + 0.5*2*h*(x[i-1]+h)*(yy[i-1] + 0.5*h*x[i-1]*y[i-1]) # improved Euler method
    ye[i] = exp(x[i]^2)
    
pe = line(Line(x,ye,{}),color='black', marker='o')    
py = line(Line(x,y,{}),color='blue', marker='o')
pyy = line(Line(x,yy,{}),color='red', marker='o')
p = pe + py + pyy
show(p)
